			// инклюдим нашу библиотеку
			#include "OneButton.h"

			///////////  пишем функции которые будут вызываться на разные типы нажатий на кнопки  ///////////////////////////

			//----   функции для первой кнопки ------------------------------------------------------------------------------

			// функция вызываеться при кратковременном нажатии на кнопку ( одиночный клик )
			void attachClick_1(void){
				printf("%s \r\n", "Click_1");
			}
			//-----------------
			// функция вызываеться при двойном кратковременном нажатии ( два коротких клика )
			void attachDoubleClick_1(void){
				printf("%s \r\n", "||DoubleClick_1");
			}
			//-----------------
			// функция вызываеться при старте зажатия и удержания кнопки ( длинное зажатие )
			void attachLongPressStart_1(void){
				printf("%s \r\n", ">>LongPressStart_1");
			}
			//-----------------
			// функция вызываеться в конце отпускания ( отжатие ) и удержания кнопки ( длинное зажатие )
			void attachLongPressStop_1(void){
				printf("%s \r\n", "<<LongPressStop_1");
			}
			//-----------------
			// функция вызываеться при удержании нажатой кнопки ( режим длинное зажатие )
			void attachDuringLongPress_1(void){
				printf("%s \r\n", "...DuringLongPress_1");
			}
			//-----------------
			//-----------------------------------------------------------------------------------------------------------------

			//----   функции для второй кнопки --------------------------------------------------------------------------------

			// функция вызываеться при кратковременном нажатии на кнопку ( одиночный клик )
			void attachClick_2(void){
				printf("%s \r\n", "Click_2");
			}
			//-----------------
			// функция вызываеться при двойном кратковременном нажатии ( два коротких клика )
			void attachDoubleClick_2(void){
				printf("%s \r\n", "||DoubleClick_2");
			}
			//-----------------
			// функция вызываеться при старте зажатия и удержания кнопки ( длинное зажатие )
			void attachLongPressStart_2(void){
				printf("%s \r\n", ">>LongPressStart_2");
			}
			//-----------------
			// функция вызываеться в конце отпускания ( отжатие ) и удержания кнопки ( длинное зажатие )
			void attachLongPressStop_2(void){
				printf("%s \r\n", "<<LongPressStop_2");
			}
			//-----------------
			// функция вызываеться при удержании нажатой кнопки ( режим длинное зажатие )
			void attachDuringLongPress_2(void){
				printf("%s \r\n", "...DuringLongPress_2");
			}
			//-----------------
			//-----------------------------------------------------------------------------------------------------------------

			///////////////////////////////////////////////////////////////////////////////////////////////////////////////////


			///////////   создаем две кнопки ( псевдо два объекта )		///////////////////////////////////////////////////////

			//--  кнопка первая настроена в режиме Pull- down ( подтянута к земле )  ---------
			oneButton bt1;

			//--  кнопка вторая настроена в режиме Pull- up ( подтянута к питанию )  ---------
			oneButton bt2;
				
			////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


void main()==================================================================================================================================================

			//////////////  инициализируем наши ( объекты ) кнопки функциями обработки   ////////////////////////////////////////////

			//-- к указателям  кнопки привязываем ссылки на функции которые должны вызываться
				
			//---  первая кнопка  -------
			bt1.clickFunc = attachClick_1;
			bt1.doubleClickFunc = attachDoubleClick_1;
			bt1.longPressStartFunc = attachLongPressStart_1;
			bt1.longPressStopFunc = attachLongPressStop_1;
			bt1.duringLongPressFunc = attachDuringLongPress_1;
			//---------------------------
			//---  вторая кнопка  -------	
			bt2.clickFunc = attachClick_2;
			bt2.doubleClickFunc = attachDoubleClick_2;
			bt2.longPressStartFunc = attachLongPressStart_2;
			bt2.longPressStopFunc = attachLongPressStop_2;
			bt2.duringLongPressFunc = attachDuringLongPress_2;
			//---------------------------

			////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////

while(1)=======================================================================================================================================================
			/////////   считываем состояние кнопок постоянно в цикле   ///////////////////////////////////////////////
			/*	 если код большой то для коректной работы чтение кнопок можно проводить при прерывании по таймеру 	*/	  
				  
				  //---  обработка первой кнопки ---- Pull-down ------------------------------
				  // при нажатии на кнопке HIGH
				  tickOneButton( &bt1, HAL_GPIO_ReadPin( GPIOA, GPIO_PIN_6 ) );
				  
				  //---  обработка второй кнопки ---- Pull-up --------------------------------
				  // при нажатии на кнопке LOW
				  tickOneButton( &bt2, !HAL_GPIO_ReadPin( GPIOA, GPIO_PIN_7 ) );
				  
				  // небольшая задержка ( можно и без нее )
				  HAL_Delay( 10 );
				  
			//////////////////////////////////////////////////////////////////////////////////////////////////////////	

=====  end while(1) =============================================================================================================================================
=====  end main() ===============================================================================================================================================
